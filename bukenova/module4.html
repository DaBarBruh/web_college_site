Задание MPIBegin31

Условие: В главном процессе дан набор из 5 чисел. Используя функцию MPI_Bcast, переслать этот набор во все подчиненные процессы и вывести в них полученные числа в том же порядке.

#include 
#pragma hdrstop
#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPIBegin31");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    
    if (rank == 0) {
        double data[5];
        for (int i = 0; i < 5; i++) {
            pt >> data[i];
        }
        MPI_Bcast(data, 5, MPI_DOUBLE, 0, MPI_COMM_WORLD);
    } else {
        double data[5];
        MPI_Bcast(data, 5, MPI_DOUBLE, 0, MPI_COMM_WORLD);
        for (int i = 0; i < 5; i++) {
            pt << data[i];
        }
    }
}

Пошаговое выполнение:

  • Инициализация MPI и получение информации о процессах
  • Главный процесс читает 5 чисел из задачника
  • Использование MPI_Bcast для рассылки данных всем процессам
  • Все процессы вызывают одну и ту же функцию MPI_Bcast
  • Подчиненные процессы получают данные и выводят их
  • Эффективная передача данных от одного ко многим

Задание MPIBegin33

Условие: В каждом процессе дан набор из 5 целых чисел. Используя функцию MPI_Gather, переслать эти наборы в главный процесс и вывести их в порядке возрастания рангов переславших их процессов.

#include 
#pragma hdrstop
#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPIBegin33");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    
    int data[5];
    int* result = new int[5 * size];
    
    for (int i = 0; i < 5; i++) {
        pt >> data[i];
    }
    
    MPI_Gather(data, 5, MPI_INT, result, 5, MPI_INT, 0, MPI_COMM_WORLD);
    
    if (rank == 0) {
        for (int i = 0; i < 5 * size; i++) {
            pt << result[i];
        }
    }
    
    delete[] result;
}

Пошаговое выполнение:

  • Инициализация MPI и получение информации о процессах
  • Каждый процесс читает свой набор из 5 чисел
  • Использование MPI_Gather для сбора данных в главном процессе
  • Главный процесс подготавливает буфер для всех данных
  • Данные автоматически упорядочиваются по рангам процессов
  • Главный процесс выводит все собранные данные

Задание MPIBegin38

Условие: В главном процессе дан набор из K(K+3)/2 целых чисел, где K — количество процессов. Используя функцию MPI_Scatterv, переслать в каждый процесс часть чисел из данного набора; при этом в процессе ранга R надо переслать R+2 очередных числа.

#include 
#pragma hdrstop
#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPIBegin38");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    
    if (rank == 0) {
        int total = size * (size + 3) / 2;
        int* data = new int[total];
        int* sendcounts = new int[size];
        int* displs = new int[size];
        
        for (int i = 0; i < total; i++) {
            pt >> data[i];
        }
        
        int index = 0;
        for (int i = 0; i < size; i++) {
            sendcounts[i] = i + 2;
            displs[i] = index;
            index += i + 2;
        }
        
        MPI_Scatterv(data, sendcounts, displs, MPI_INT, NULL, 0, MPI_INT, 0, MPI_COMM_WORLD);
        delete[] data;
        delete[] sendcounts;
        delete[] displs;
    } else {
        int count = rank + 2;
        int* recvdata = new int[count];
        MPI_Scatterv(NULL, NULL, NULL, MPI_INT, recvdata, count, MPI_INT, 0, MPI_COMM_WORLD);
        
        for (int i = 0; i < count; i++) {
            pt << recvdata[i];
        }
        delete[] recvdata;
    }
}

Пошаговое выполнение:

  • Инициализация MPI и получение информации о процессах
  • Главный процесс вычисляет общее количество чисел
  • Подготовка массивов sendcounts и displs для нерегулярного рассеивания
  • Процесс ранга R получает R+2 чисел
  • Использование MPI_Scatterv для нерегулярного распределения данных
  • Каждый процесс выводит полученные данные
